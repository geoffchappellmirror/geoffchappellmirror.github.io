ATL Attributes: rdx

Creates a registry key or modifies an existing registry key

Syntax 

[
    define_attribute = "rdx",
    argument_list = {
        "key",
        "valuename",
        "regtype"
    },
    required_arguments = {
        "key",
        "regtype"
    },
    check_argument ("regtype") = {
        "text",
        "dword",
        "binary",
        "CString"
    },
    help_string = "Creates a registry key or modifies an existing registry key",
    help_string ("key") = "Name of the key to be created or opened",
    help_string ("valuename") = "Specifies the value field to be set",
    help_string ("regtype") = "Type of registry key being added",
    usage = eCoClassMemberUsage | eMemberUsage,
    group = eRegistryGroup,
    shipping = true,
    variable_args = true
];
