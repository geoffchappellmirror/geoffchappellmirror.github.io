C++ Attributes: implements

Specifies (disp)interfaces that are forced to be members of the IDL coclass

Syntax 

[
    define_attribute = "implements",
    argument_list = {
        "interfaces",
        "dispinterfaces"
    },
    default_value ("interfaces") = "",
    default_value ("dispinterfaces") = "",
    help_string = "Specifies (disp)interfaces that are forced to be members of the IDL coclass",
    usage = eCoClassUsage,
    group = eCompilerGroup,
    repeatable = "true",
    multivalue ("interfaces") = "true",
    multivalue ("dispinterfaces") = "true",
    shipping = "true"
];
