ATL Attributes: db_source

Creates and encapsulates a connection, through a provider, to a data source

Syntax 

[
    define_attribute = "db_source",
    argument_list = {
        "db_source",
        "name",
        "hresult"
    },
    required_arguments = "db_source",
    help_string = "Creates and encapsulates a connection, through a provider, to a data source",
    help_string ("db_source") = "Connection string for opening the data source",
    help_string ("name") = "Name of a local variable that can be used to access the data source",
    usage = eClassUsage | eCoClassUsage | eCoClassMemberUsage | eCoClassMethodUsage | eLocalUsage | eMemberUsage | eMethodUsage | eStructUsage,
    group = eDBConsumerGroup,
    shipping = true,
    variable_args = true
];
