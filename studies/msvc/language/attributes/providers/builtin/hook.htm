C++ Attributes: hook

The hook attribute is overloaded. Depending on which arguments are given, it

  • Hooks an event handler to an event
  • Hooks all event handlers to an event for a given receiver

Syntax 

[
    define_attribute = "hook",
    argument_list = {
        "event",
        "source",
        "handler",
        "receiver"
    },
    default_value ("receiver") = "this",
    help_string = "Hooks an event handler to an event",
    usage = eLocalUsage,
    group = eCompilerGroup,
    shipping = "true",
    overload = "true"
];
[
    define_attribute = "hook",
    argument_list = {
        "event",
        "source",
        "receiver"
    },
    default_value ("receiver") = "this",
    help_string = "Hooks all event handlers to an event for a given receiver",
    usage = eLocalUsage,
    group = eCompilerGroup,
    shipping = "true",
    overload = "true"
];
